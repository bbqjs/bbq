/**
 * GUIButton
 * 
 * This class creates a standards compliant button that should be used in conjunction with a ButtonHolder object.  It
 * is based around the HTML list element an contains an anchor tag.  It supports the following options:
 * 
 */

/**
 * Supports the following options: 
 * 
 * options: {
 * 		buttonText: String
 * 		startDisabled: boolean
 * 		startDown: boolean
 * 		rememberDownState: boolean
 * 		toggleButton: boolean
 * 		onclick: Function
 * 		onmouseout: Function
 * 		onmouseover: Function
 * 		onmousedown: Function
 * 		onmouseup: Function
 * }
 */
behaviour.gui.button.GUIButton.prototype.initialize = function(options) {
	var parentConstructor = behaviour.gui.GUIWidget.prototype.initialize;
	parentConstructor.apply(this, arguments);
	
	this.setRootNode(document.createElement("li"));
	this.addClass("GUIButton");
	
	this.buttonLink = DOMUtil.createTextElement("a", (this.options.buttonText ? this.options.buttonText : " "), {href: ".", onclick: function(){this.blur(); return false}, title: (options.buttonText ? options.buttonText : "")});
	this.appendChild(this.buttonLink);
	
	if(this.options.startDisabled) {
		this.setDisabled(true);
	} else {
		this.setDisabled(false);
	}
	
	if(this.options.startDown) {
		this.setDown(true);
	} else {
		this.setDown(false);
	}
	
	Event.observe(this.buttonLink, "click", this.buttonClicked.bindAsEventListener(this));
	Event.observe(this.buttonLink, "mouseout", this.mouseOut.bindAsEventListener(this));
	Event.observe(this.buttonLink, "mouseover", this.mouseOver.bindAsEventListener(this));
	Event.observe(this.buttonLink, "mousedown", this.mouseDown.bindAsEventListener(this));
	Event.observe(this.buttonLink, "mouseup", this.mouseUp.bindAsEventListener(this));
}

behaviour.gui.button.GUIButton.prototype.buttonClicked = function(event) {
	this.buttonHolder._buttonClicked(this);
	
	if(this._processCallback("onclick")) {
		if(this.buttonHolder) {
			this.buttonHolder.clearDown();
		}
		
		if(this.options.rememberDownState) {
			this.setDown(true);
		}
		
		if(this.options.toggleButton) {
			this.setDown(this.buttonIsDown);			
		}
	}
}

behaviour.gui.button.GUIButton.prototype.mouseOut = function(event) {
	if(!this.disabled) {
		this.removeClass("buttonOver");
		this._processCallback("onmouseout");
	}
}

behaviour.gui.button.GUIButton.prototype.mouseOver = function(event) {
	if(!this.disabled) {
		this.addClass("buttonOver");
		this._processCallback("onmouseover");
	}
}

behaviour.gui.button.GUIButton.prototype.mouseDown = function(event) {
	if(!this.disabled) {
		this.addClass("buttonDown");
		this._processCallback("onmousedown");
	}
}

behaviour.gui.button.GUIButton.prototype.mouseUp = function(event) {
	if(!this.disabled) {
		this.removeClass("buttonDown");
		this._processCallback("onmouseup");
	}
}

/**
 * Calls any callback method if defined and the button is enabled.
 * 
 * @return	boolean	true if we were able to callback, false otherwise
 */
behaviour.gui.button.GUIButton.prototype._processCallback = function(callback) {
	if(!this.disabled && this.options[callback] && this.options[callback] instanceof Function) {
		var args = new Array();
		
		if(this.options["callbackObject"]) {
			args.push(this.options["callbackObject"]);
		}
		
		this.options[callback].apply(this, args);
		
		return true;
	}
	
	return false;
}

behaviour.gui.button.GUIButton.prototype.setDown = function(down) {
	this.buttonIsDown = down;
	
	if(this.buttonIsDown) {
		this.buttonHolder.setSelectedIndex(this.getIndex());
		this.addClass("buttonDown");
	} else {
		this.removeClass("buttonDown");
	}
}

behaviour.gui.button.GUIButton.prototype.clearDown = function() {
	this.removeClass("buttonDown");
}

behaviour.gui.button.GUIButton.prototype.isDown = function() {
	return this.buttonIsDown;
}

behaviour.gui.button.GUIButton.prototype.setDisabled = function(disabled) {
	this.disabled = disabled;
	this[(this.disabled ? "add" : "remove") + "Class"]("buttonDisabled");
	this[(this.disabled ? "remove" : "add") + "Class"]("buttonEnabled");
}

behaviour.gui.button.GUIButton.prototype.setOnClick = function(callback) {
	this.options["onclick"] = callback;
}

behaviour.gui.button.GUIButton.prototype.getIndex = function() {
	return this._index;
}

behaviour.gui.button.GUIButton.prototype.setIndex = function(index) {
	this._index = index;
}
