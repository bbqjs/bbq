behaviour.gui.updateables.UpdateableTextArea = new Class.create();

Object.extend(behaviour.gui.updateables.UpdateableTextArea.prototype, behaviour.gui.updateables.UpdateableField.prototype);

/**
 * @param	{Object}									An options object
 */
behaviour.gui.updateables.UpdateableTextArea.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.updateables.UpdateableField.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	this.addClass("UpdateableTextArea");
}

behaviour.gui.updateables.UpdateableTextArea.prototype.createEditField = function() {
	Log.info("creating edit field");
	this.empty();
	this.appendChild(DOMUtil.createTextElement("textarea", this.currentValue, {disabled: this.disabled, onchange: this.updateLocalValue.bindAsEventListener(this)}));
	Log.info("created edit field");
}

behaviour.gui.updateables.UpdateableTextArea.prototype.createViewField = function() {
	this.empty();
	
	var theText = this.displayValue.split("\n");
	
	for(var i = 0; i < theText.length; i++) {
		this.appendChild(document.createTextNode(theText[i]));
		
		if(theText.length > 1 && i != (theText.length - 1)) {
			this.appendChild(document.createElement("br"));
		}
	}
}
