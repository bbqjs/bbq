behaviour.gui.updateables.UpdateableDropDown = new Class.create();

// extends TitleBar
Object.extend(behaviour.gui.updateables.UpdateableDropDown.prototype, behaviour.gui.updateables.UpdateableField.prototype);

behaviour.gui.updateables.UpdateableDropDown.prototype.dropDown;
behaviour.gui.updateables.UpdateableDropDown.prototype.fields;

/**
 * Supports the folloing options:
 * 
 * options: {
 * 		keys: Array
 * }
 */
behaviour.gui.updateables.UpdateableDropDown.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.updateables.UpdateableField.prototype.initialize;
	parentMethod.apply(this, arguments);	
	
	this.fields = this.options.keys;
	this.originalValue = this.options.initialValue ? this.options.initialValue : 0;
	this.currentValue = this.originalValue;
	this.displayValue = this.fields[this.currentValue];
	this.addClass("UpdateableDropDown");
}

behaviour.gui.updateables.UpdateableDropDown.prototype.cancelEdit = function() {
	this.currentValue = this.originalValue;
	this.displayValue = this.fields[this.currentValue];
}

behaviour.gui.updateables.UpdateableDropDown.prototype.createEditField = function() {
	this.dropDown = document.createElement("select");
	this.dropDown.onchange = this.updateLocalValue.bindAsEventListener(this);
	
	for(var i = 0; i < this.fields.length; i++) {
		var node = document.createElement("option");
		node.appendChild(document.createTextNode(this.fields[i]));
		node.value = i;
		
		if(i == this.currentValue) {
			node.selected = "selected";
		}
		
		this.dropDown.appendChild(node);
	}
	
	this.dropDown.disabled = this.disabled;
	return this.dropDown;
}

behaviour.gui.updateables.UpdateableDropDown.prototype.saveEdit = function() {
	this.originalValue = this.currentValue;
	this.displayValue = this.fields[this.currentValue];
}

behaviour.gui.updateables.UpdateableDropDown.prototype.updateLocalValue = function() {
	if(this.dropDown && this.dropDown.selectedIndex != null) {
		this.currentValue = this.dropDown.selectedIndex;
		this.displayValue = this.fields[this.currentValue];
	}
}
