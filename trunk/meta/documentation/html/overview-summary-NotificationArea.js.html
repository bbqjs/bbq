behaviour.gui.notificationArea.NotificationArea.prototype.initialize = function(controller)
	{
	var parentMethod = behaviour.gui.GUIWidget.prototype.initialize;
	parentMethod.call(this, controller);
	
	this.rootNode = document.createElement("div");
	this.controller.domUtil.addClass(this.rootNode, "notificationArea");
	this.rootNode.style.display = "none";
	
	this.innerDiv = document.createElement("div");
	this.controller.domUtil.addClass(this.innerDiv, "innerDiv");
	this.rootNode.appendChild(this.innerDiv);
	
	this.active = false;
	
	this.messageArea = new behaviour.gui.notificationArea.MessageArea(this.controller);
	this.loadingArea = new behaviour.gui.notificationArea.LoadingArea(this.controller);
		
	var table = document.createElement("table");
	var tableBody = document.createElement("tbody");
	var tableRow = document.createElement("tr");
	
	table.appendChild(tableBody);
	tableBody.appendChild(tableRow);
	
	this.loadingArea.appendTo(tableRow);
	this.messageArea.appendTo(tableRow);
	
	this.innerDiv.appendChild(table);
	}

behaviour.gui.notificationArea.NotificationArea.prototype.startLoad = function()
	{
	this.loadingArea.startLoad();
	this.appear();
	}

behaviour.gui.notificationArea.NotificationArea.prototype.stopLoad = function()
	{
	this.loadingArea.stopLoad();
	}

behaviour.gui.notificationArea.NotificationArea.prototype.setMessage = function(messageHeader, messageText, messageStatus)
	{
	this.messageArea.setMessage(messageHeader, messageText, messageStatus);
	this.appear();
	}

behaviour.gui.notificationArea.NotificationArea.prototype.watchChildren = function()
	{
	if(!this.loadingArea.active && !this.messageArea.active)
		{
		Log.info("Done watching stuff");
		this.active = false;
		clearInterval(this.watcherInterval);
		Effect.Fade(this.rootNode, {duration: 0.5});
		}
	}

behaviour.gui.notificationArea.NotificationArea.prototype.appear = function()
	{
	if(this.active == false && (this.loadingArea.active || this.messageArea.active))
			{
			this.active = true;
			this.watcherInterval = setInterval(this.watchChildren.bind(this), 500);
			Effect.Appear(this.rootNode, {duration: 0.25});
			Log.info("Watching stuff");
			}
	}
