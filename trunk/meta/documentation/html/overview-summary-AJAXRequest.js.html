/**
 * Wrapper for Prototype Ajax class.
 * 
 * Adds extras such as an independant server timeout and error handling
 */
behaviour.ajax.AJAXRequest = Class.create();

/**
 * Constructor
 * 
 * @param	Object	options	The URL to send the request to
 * 
 * Supports the following options:
 * 
 * options {
 * 		url:	String								// where to send the request to
 * 		method: String						// post or get
 * 		args: Object							// key->value pairs to convert to query string
 * 		onsuccess: Function
 * 		onfaliure: Function
 * 		onexception: Function
 * }
 */
behaviour.ajax.AJAXRequest.prototype.initialize = function(options) {
	this.options = options;
	
	if(!this.options.method) {
		this.options.method = "post";
	}
	
	if(!this.options.args) {
		this.options.args = {};
	}
	
	this.sendRequest();
}

behaviour.ajax.AJAXRequest.prototype._processArgs = function(string, object) {
	if(!object) {
		return string;
	} else {
		var numArgs = object.length;
		var i = 0;
		
		for(var key in object) {
			string += this._processArg(key, object[key]);
			
			if(i < object.length) {
				string += "&";
			}
			
			i++;
		}
	}
}

behaviour.ajax.AJAXRequest.prototype._processArg = function(key, value) {
	var output = "";
	
	if(value instanceof Array) {
		for(var i = 0; i < value.length; i++) {
			output += key + "[]" + value[i];
			
			if(i < value.length) {
				output += "&";
			}
		}
	} else {
		output = key + "=" + value;
	}
	
	return output;
}

/**
 * Sends the request via the specified method
 * 
 * @return	void
 */
behaviour.ajax.AJAXRequest.prototype.sendRequest = function() {
	new Ajax.Request(this.options.url, {
		method: this.options.method, 
		parameters: this.options.args, 
		onSuccess: this.onSuccess.bind(this), 
		onFailure: this.onFailure.bind(this), 
		onExcepton: this.onException.bind(this)
	});
	
	// get time out data from server configuration
	if(typeof(ServerConfig) != "undefined" && ServerConfig["scripttimeout"]) {
		this.timeOut = ServerConfig["scripttimeout"];
	} else {
		this.timeOut = 30;
	}
	
	// check timeout every second
	this.interval = setInterval(this.checkTimeOut.bind(this), 1000);
	
	if(typeof(NotificationArea) != "undefined") {
		NotificationArea.startLoad();
	}
}

/**
 * @access	protected
 */
behaviour.ajax.AJAXRequest.prototype._callHandler = function(handlerName, args) {
	if(typeof(NotificationArea) != "undefined") {
		NotificationArea.stopLoad();
	}
	
	clearInterval(this.interval);
	
	if(this.options[handlerName] && this.options[handlerName] instanceof Function) {
		this.options[handlerName].apply(this, args);
	} else if(this.options["onAnything"] && this.options["onAnything"] instanceof Function) {
		this.options["onAnything"].apply(this, args);
	}
}

/**
 * If we are not Firefox will enter a log line into Firebug Lite.
 * Firefox picks this stuff up automatically so it's not required.
 * 
 * @private
 */
behaviour.ajax.AJAXRequest.prototype._log = function(type, message) {
	if(Browser && !Browser.Mozilla) {
		if(message instanceof Array) {
			console.group();
			message.each(function(element, index){Log[type](element);});
			console.groupEnd();
		} else {
			Log[type](message);
		}
	}
}

behaviour.ajax.AJAXRequest.prototype.onSuccess = function(serverResponse) {
	this._log('info', [this.options.method.toUpperCase() + ' ' + this.options.url, serverResponse.responseText]);
	this._callHandler("onSuccess", $A(arguments));
}

behaviour.ajax.AJAXRequest.prototype.onFailure = function() {
	this._log('warn', 'Request to ' + this.options.method.toUpperCase() + ' ' + this.options.url + " failed");
	this._callHandler("onFaliure", $A(arguments));
}

behaviour.ajax.AJAXRequest.prototype.onException = function() {
	this._log('warn', 'Request to ' + this.options.method.toUpperCase() + ' ' + this.options.url + " threw exception");
	this._callHandler("onException", $A(arguments));
}

/**
 * Checks to see if the timer has reached 0.  If so the server request has timed out.
 * 
 * @return	void
 */
behaviour.ajax.AJAXRequest.prototype.checkTimeOut = function() {
	if(this.timeOut == 0) {
		this.timedOut();
		clearInterval(this.interval);
	} else {
		this.timeOut--;
	}
}

/**
 * Shows the user a warning
 * 
 * @return	void
 */
behaviour.ajax.AJAXRequest.prototype.timedOut = function() {
	Log.warn("Ajax call to " + this.url + " timed out");
	
	if(typeof(NotificationArea) != "undefined") {
		NotificationArea.notificationArea.stopLoad();
		NotificationArea.notificationArea.setMessage(Language.get("generic", "errors", "ajaxtimeout", "header"), Language.get("generic", "errors", "ajaxtimeout", "message"), "error");
		NotificationArea.notificationArea.setMessage(Language.get("generic", "errors", "reloadpage", "header"), Language.get("generic", "errors", "reloadpage", "message"), "error");
	}
}
