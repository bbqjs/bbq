// constructor
behaviour.ajax.BackButtonFixer.prototype.initialize = function(controller, emptyMethod) {
	// the object to call methods on
	this.controller = controller;
	
	// the method to call to return the page to it's initial state
	if(emptyMethod) {
		this.emptyMethod = emptyMethod;
	}
	
	// have we been loaded with a method request
	this.lastMethod = this.getMethod(document.location.href, true);
	
	// stupid IE
	this.neediFrameFix = Browser.InternetExplorer ? true : false;
	
	if(this.neediFrameFix) {
		this.createiFrame(this.lastMethod);
	}

	// call the requested (bookmarked) method
	this.overrideMethodCheck = false;

	if(this.lastMethod != null) {
		this.overrideMethodCheck = true;
		this.callMethod(this.lastMethod);
	}
	
	// start checking to see if the URL has changed
	setInterval(this.checkLocation.bind(this), 100);
}

// creates an iFrame to add to the page
behaviour.ajax.BackButtonFixer.prototype.createiFrame = function(method) {
	var iframe = document.createElement("iframe");
	iframe.id = "iFrameFix";
	iframe.name = "iFrameFix";
	iframe.style.display = "none";
	document.getElementsByTagName("body")[0].appendChild(iframe);
	this.loadiFrameURL(method);
}

// looks at the URL in the address bar
behaviour.ajax.BackButtonFixer.prototype.checkLocation = function() {
	var method = this.getMethod();
	
	if(this.lastMethod != method) { // new method call
		//alert("method = " + method + " lastMethod = " + this.lastMethod);
		
		if(this.neediFrameFix) {
			this.loadiFrameURL(method);
		} else {
			this.callMethod(method);
		}
	}
}

// gets the method name from the passed argument
behaviour.ajax.BackButtonFixer.prototype.getMethod = function(href, acceptNullValue) {
	if(!href) {
		href = document.location.href;
	}

	var method = href.split("#");
	
	if(method.length > 1) {
		return method[1];
	}

	return acceptNullValue ? null : "";
}

// loads the iFrame for crap browsers or calls the callMethod method
behaviour.ajax.BackButtonFixer.prototype.loadiFrameURL = function(method) {
	$("iFrameFix").src = this.iFrameURL + "?time="+(new Date()).getTime() + (method != null ? "#" + method : "");
}

// tells the iFrame to examine it's URL and tell us what the method name is
behaviour.ajax.BackButtonFixer.prototype.iFrameLoaded = function(frameWindow) {
	//alert("iframe callback");
	frameWindow.owner = this;
	frameWindow.readURLandCallMethod();
}

// calls the passed method name - override this method
behaviour.ajax.BackButtonFixer.prototype.callMethod = function(passed) {
	//alert("calling " + passed + ", last method was " + this.lastMethod);
	
	if(passed) {
		var query = passed.split("/");
		var method = query[0];
		var args = new Array();

		// don't call the same method twice
		if(!this.overrideMethodCheck && method == this.lastMethod) {
			this.overrideMethodCheck = false;
			return;
		}

		this.lastMethod = passed;

		// build arguments
		for(var i = 1; i < query.length; i++) {
			args[(i-1)] = query[i];
		}
		
		// make sure method exists
		if(this.controller[method]) {
			//try
			//	{
				// call the method
				//Log.info("Calling method " + method);
				var methodToCall = this.controller[method].bind(this.controller);
				methodToCall.apply(this, args);
				//Log.info("Called method " + method);
			//	}
			//catch(e)
			//	{
			//	Log.error("Error calling method " + method + " passed as part of " + passed + " - " + e);
			//	}
		} else { // tried to call non-existant method
			//Log.info("Method  " + method + " not found on controller");
		}
	} else if(this.emptyMethod) {
		//Log.info("Called empty method");
		this.emptyMethod.call();
		this.lastMethod = "";
	}
}
