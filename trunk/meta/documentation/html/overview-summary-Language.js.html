/* Holds language information */

behaviour.util.Language = new Class.create();

behaviour.util.Language.prototype.controller;
behaviour.util.Language.prototype.moduleName;
behaviour.util.Language.prototype.lang;

behaviour.util.Language.prototype.initialize = function(module, controller) {
	this.controller = controller;
	this.controllerCallback = this.controller.languageLoaded.bind(this.controller);
	this.moduleName = module;
	this.lang = new behaviour.ajax.XMLArray(this.controller, "d39a80445e3d667000f8abf19e640bbe", this.constructed.bind(this));
	this.lang.sendRequest('get');
}

// calls a callback method on the controller
behaviour.util.Language.prototype.constructed = function(reply) {
	Log.info("Language loaded");
	this.controllerCallback();
}


// returns a requested bit of language
behaviour.util.Language.prototype.get = function() {
	if(this.lang && this.lang.data) {
		var methodToCall = this.lang.get.bind(this.lang);
		return methodToCall.apply(this, arguments);
	} else {
		Log.error("Call to language.get when it's not yet loaded");
	}
}
