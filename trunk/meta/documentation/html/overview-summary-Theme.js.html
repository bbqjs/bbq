behaviour.gui.Theme = new Class.create();

behaviour.gui.Theme.prototype.initialize = function(theme) {
	if(theme) {
		this.setTheme(theme);
	}
}

behaviour.gui.Theme.prototype.setTheme = function(themeType) {
	this.removeThemes();
			
	var themes = new Array();
	themes[0] = "coffee-light";
	themes[1] = "coffee-dark";
	themes[2] = "win-aqua";
	themes[3] = "mac-osx";
			
	var theme = themes[0];
	
	if(themeType) {
		theme = themes[themeType];
	}
			
	var styleNode = document.createElement("link");
	styleNode.href = "/css/theme/" + theme + "/theme.css";
	styleNode.rel = "stylesheet";
	styleNode.type = "text/css";
			
	var documentHead = $$("head")[0];
	documentHead.appendChild(styleNode);
}
		
behaviour.gui.Theme.prototype.removeThemes = function() {
	var documentHead = $$("head")[0];
	var styleNodes = $$("link");
	
	for(var i = 0; i < styleNodes.length; i++) {
		if(styleNodes[i].href && !styleNodes[i].href.match(/base.css/)) {
			documentHead.removeChild(styleNodes[i]);
		}
	}
}
