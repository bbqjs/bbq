behaviour.gui.notificationArea.MessageArea = new Class.create();

// extends GUIWidget
Object.extend(behaviour.gui.notificationArea.MessageArea.prototype, behaviour.gui.GUIWidget.prototype);

behaviour.gui.notificationArea.MessageArea.prototype.controller;
behaviour.gui.notificationArea.MessageArea.prototype.messageQueue;
behaviour.gui.notificationArea.MessageArea.prototype.active;

behaviour.gui.notificationArea.MessageArea.prototype.initialize = function(controller)
	{
	var parentMethod = behaviour.gui.GUIWidget.prototype.initialize;
	parentMethod.call(this, controller);
		
	this.rootNode = document.createElement("td");
	this.rootNode.id = "notificationMessageArea";
	this.rootNode.style.display = "none";
	
	this.active = false;
	this.messageQueue = new Array();
	}

behaviour.gui.notificationArea.MessageArea.prototype.setMessage = function(messageHeader, messageText, messageStatus)
	{
	var message = new Object;
	
	message["header"] = messageHeader;
	message["text"] = messageText;
	message["status"] = messageStatus;
	
	this.messageQueue[this.messageQueue.length] = message;
	
	if(!this.active)
		{
		this.showMessage();
		}
	}

behaviour.gui.notificationArea.MessageArea.prototype.showMessage = function()
	{
	if(this.active)
		{
		if(this.messageQueue.length == 0)
			{
			this.active = false;
			Effect.Fade(this.rootNode, {duration: 0.25});
			
			setTimeout(this.clearMessage.bind(this), 255);
			return;
			}
		
		// fade out old message and show new one
		Effect.Fade(this.rootNode, {duration: 0.25});
		setTimeout(this.showNextMessage.bind(this), 250);
		}
	else // construct display and fade in
		{
		this.active = true;
		this.buildMessage();
		Effect.Appear(this.rootNode, {duration: 0.5});
		}
	
	setTimeout(this.showMessage.bind(this), 5000);
	}

behaviour.gui.notificationArea.MessageArea.prototype.showNextMessage = function()
	{
	this.buildMessage();
	Effect.Appear(this.rootNode, {duration: 0.25});
	}

behaviour.gui.notificationArea.MessageArea.prototype.buildMessage = function()
	{
	var message = this.messageQueue.shift();
	
	Log.info("Showing message: " + message["header"]);
	
	var output = document.createDocumentFragment();
	
	var header = document.createElement("h2");
	header.appendChild(document.createTextNode(message["header"]));
	output.appendChild(header);
	
	var messageText = document.createElement("p");
	messageText.appendChild(document.createTextNode(message["text"]));
	output.appendChild(messageText);
	
	this.rootNode.className = "";
	this.controller.domUtil.addClass(this.rootNode, message["status"]);
	
	this.controller.domUtil.emptyNode(this.rootNode);
	this.rootNode.appendChild(output);
	}

behaviour.gui.notificationArea.MessageArea.prototype.clearMessage = function()
	{
	this.controller.domUtil.emptyNode(this.rootNode);
	}
