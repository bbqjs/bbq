behaviour.gui.updateables.UpdateableAddress.prototype.initialize = function(passedValue) {
	var startValue = new Array(null, "", "", "", "", "", User["country"]);
	var inputFieldNames = new Array("addressCategory", "house", "street", "area", "region", "postcode", "country");
	
	var parentConstructor = behaviour.gui.updateables.CompoundUpdateableField.prototype.initialize;
	parentConstructor.call(this, passedValue, startValue, inputFieldNames);
	
	DOMUtil.addClass(this.rootNode, "addressField");
	
	this.inputFields = new Array(7);
	this.inputFields[0] = new behaviour.gui.updateables.UpdateableDropDown(this.currentValue[0], Language.getArray(3, "generic",  "contact", "category-"));
	this.inputFields[1] = new behaviour.gui.updateables.UpdateableTextField(this.currentValue[1]);
	this.inputFields[2] = new behaviour.gui.updateables.UpdateableTextField(this.currentValue[2]);
	this.inputFields[3] = new behaviour.gui.updateables.UpdateableTextField(this.currentValue[3]);
	this.inputFields[4] = new behaviour.gui.updateables.UpdateableTextField(this.currentValue[4]);
	this.inputFields[5] = new behaviour.gui.updateables.UpdateableTextField(this.currentValue[5]);
	this.inputFields[6] = new behaviour.gui.updateables.UpdateableDropDown(this.currentValue[6], Language.getArray(240, "generic",  "countries", "country-"));
		
	this.constructViewNode();
}
