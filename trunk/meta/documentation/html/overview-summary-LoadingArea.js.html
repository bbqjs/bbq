behaviour.gui.notificationArea.LoadingArea = new Class.create();

// extends GUIWidget
Object.extend(behaviour.gui.notificationArea.LoadingArea.prototype, behaviour.gui.GUIWidget.prototype);

behaviour.gui.notificationArea.LoadingArea.prototype.initialize = function(controller)
	{
	var parentMethod = behaviour.gui.GUIWidget.prototype.initialize;
	parentMethod.call(this, controller);
		
	this.rootNode = document.createElement("td");
	this.rootNode.id = "notificationLoadingArea";
	this.rootNode.style.display = "none";
	
	this.active = false;
	this.numLoading = 0;
	this.numCompleted = 0;
	}

behaviour.gui.notificationArea.LoadingArea.prototype.startLoad = function()
	{
	this.numLoading++;
	this.render();
	
	if(this.numLoading == 1)
		{
		this.active = true;
		Effect.Appear(this.rootNode, {duration: 0.1});
		}
	}

behaviour.gui.notificationArea.LoadingArea.prototype.stopLoad = function()
	{
	this.numLoading--;
	this.numCompleted++;
	
	if(this.numLoading == 0)
		{
		this.active = false;
		this.numCompleted = 0;
		Effect.Fade(this.rootNode, {duration: 0.6});
		setTimeout(this.clearLoading.bind(this), 600);
		return;
		}
	
	this.render();
	}

behaviour.gui.notificationArea.LoadingArea.prototype.render = function()
	{
	this.clearLoading();
	
	Log.info("Num loading = " + this.numLoading);
	
	var output = document.createDocumentFragment();
	
	var header = document.createElement("h2");
	header.appendChild(document.createTextNode(this.controller.getLanguage("generic", "loading", "header")));
	output.appendChild(header);
	
	if(this.numLoading == 1)
		{
		var loadingPara = document.createElement("p");
		loadingPara.appendChild(document.createTextNode(this.controller.getLanguage("generic", "loading", "paragraph")));
		output.appendChild(loadingPara);
		
		this.rootNode.appendChild(output);
		}
	else if(this.numLoading > 1)
		{
		var loadingPara = document.createElement("p");
		loadingPara.appendChild(document.createTextNode(this.controller.getLanguage("generic", "loading", "paragraph")));
		output.appendChild(loadingPara);	

		this.rootNode.appendChild(output);
		}
	}

behaviour.gui.notificationArea.LoadingArea.prototype.clearLoading = function()
	{
	this.controller.domUtil.emptyNode(this.rootNode);
	}
