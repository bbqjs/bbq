behaviour.gui.DatePicker = new Class.create();

Object.extend(behaviour.gui.DatePicker.prototype, behaviour.gui.GUIWidget.prototype);

// Constructor
behaviour.gui.DatePicker.prototype.initialize = function(controller, dateController, title, startDate, startX, startY)
	{
	this.controller = controller;
	this.dateController = dateController;
		
	if(startDate instanceof Date)
		{
		this.date = startDate;
		}
	else
		{
		this.date = new Date(startDate);
		}
	
	this.floatingWindow = new behaviour.gui.FloatingWindow(this.controller, this, startX, startY)
	this.calendar = new behaviour.gui.PickableCalendar(controller, this, title, startDate, startX, startY)
	this.floatingWindow.setContent(this.calendar);
	this.floatingWindow.appendTo(document.getElementsByTagName("body")[0]);
	}

behaviour.gui.DatePicker.prototype.render = function()
	{
	this.floatingWindow.render();
	}

behaviour.gui.DatePicker.prototype.datePicked = function(date)
	{
	this.floatingWindow.closeWindow();
	this.dateController.datePicked(date);
	}

behaviour.gui.DatePicker.prototype.windowClosed = function()
	{
	this.dateController.datePicked();
	}
