/**
 * @class
 * 
 * UK dates
 */
behaviour.date.DateFormat1 = new Class.create();

Object.extend(behaviour.date.DateFormat1.prototype, behaviour.date.DateFormat.prototype);

behaviour.date.DateFormat1.prototype._dateSeperator = "/";
behaviour.date.DateFormat1.prototype._timeSeperator = ":";

behaviour.date.DateFormat1.prototype.toString = function(options) {
	return this.getLongDate(options);
}

behaviour.date.DateFormat1.prototype.getLongDate = function(options) {
	return this.getDate() + " " + this.getTime(options);
}

behaviour.date.DateFormat1.prototype.getDate = function(options) {
	return this._date.getDate() + this._dateSeperator + (this._date.getMonth() + 1) + this._dateSeperator + this._date.getFullYear();
}

behaviour.date.DateFormat1.prototype.getTime = function(options) {
	return this._date.getHours() + this._timeSeperator + (this._date.getMinutes() < 10 ? "0" + this._date.getMinutes() : this._date.getMinutes());
}

behaviour.date.DateFormat1.prototype.getOrder = function() {
	return new Array("Day", "DateSeperator", "Month", "DateSeperator", "Year", "EmptySpace", "Hour", "TimeSeperator", "Minute");
}
