/**

 * UpdateableTextField
 * 
 * A single line HTML text input field.
 */
behaviour.gui.updateables.UpdateableTextField = new Class.create();

/**
 * @extends	behaviour.gui.updateables.UpdateableField
 */
Object.extend(behaviour.gui.updateables.UpdateableTextField.prototype, behaviour.gui.updateables.UpdateableField.prototype);

behaviour.gui.updateables.UpdateableTextField.prototype.inputField;

/**
 * @param	{Object}									An options object
 */
behaviour.gui.updateables.UpdateableTextField.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.updateables.UpdateableField.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	this.addClass("UpdateableTextField");
}

/**
 * Creates an editable representation of this field
 */
behaviour.gui.updateables.UpdateableTextField.prototype.createEditField = function() {
	this.inputField = DOMUtil.createElement("input", {
		type: "text",
		value: this.displayValue,
		onchange: this.updateLocalValue.bindAsEventListener(this),
		onkeydown: this.preUpdateLocalValue.bindAsEventListener(this),
		disabled: this.disabled,
		autocomplete: "off",
		oncomplete: "off"
	});
	
	return this.inputField;
}

/**
 * Sets a timeout to call updateLocalValue - necessary as Internet Explorer is stupid
 * 
 * @return	void
 */
behaviour.gui.updateables.UpdateableTextField.prototype.preUpdateLocalValue = function(event) {
	setTimeout(this.updateLocalValue.bind(this), 100);
}

/**
 * Updates the local store of the field value and makes a fuss if we are required but don't have a value.
 * 
 * @return	void
 */
behaviour.gui.updateables.UpdateableTextField.prototype.updateLocalValue = function() {
	//Log.info("Updating local value - '" + this.currentValue + "' becoming '" + this.rootNode.value + "'");
	
	this.currentValue = this.inputField.value;
	this.checkForRequired();
}
