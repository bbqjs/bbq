behaviour.gui.updateables.CompoundUpdateableField.prototype.initialize = function(options) {
	if(options.passedValue) {
		options.startValue = new Array();
		var i = 0;
		
		for(var key in passedValue) {
			//Log.info(key + " = " + passedValue[key]);
			options.startValue[i] = passedValue[key];
			i++;
		}
	}
	
	this.deleted = false;
	
	this.rootNode = document.createElement("span");
	DOMUtil.addClass(this.rootNode, "compoundUpdateableField");
	
	this.editMode = false;
	this.currentValue = options.startValue;
	this.originalValue = options.startValue;
	this.inputFieldNames = options.inputFieldNames;
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.createEditField = function() {
	if(this.deleted) {
		return;
	}
	
	DOMUtil.emptyNode(this.rootNode);
	var form = DOMUtil.createNullForm();
	
	//this.deleteButton = DOMUtil.createSubmitInputNode(null, null, "compoundFieldDeleteButton", Language.get("generic",  "delete"));
	//this.deleteButton = DOMUtil.createSubmitInputNode(null, null, "compoundFieldDeleteButton", "-");
	this.deleteButton = DOMUtil.createTextElement("a", Language.get("generic",  "delete"), {href: ".", onclick: this.deleteField.bindAsEventListener(this), className: "deleteLink"});
	form.appendChild(this.deleteButton);
	
	for(var i = 0; i < this.inputFields.length; i++) {
		this.inputFields[i].setEditMode(true);
		this.inputFields[i].appendTo(form);
	}
	
	this.rootNode.appendChild(form);
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.createViewField = function() {
	if(this.deleted) {
		return;
	}
	
	this.constructViewNode();
	DOMUtil.emptyNode(this.rootNode);
	this.rootNode.appendChild(this.displayValue);
	
	//Log.info("created view field");
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.constructViewNode = function() {
	this.displayValue = document.createDocumentFragment();
	
	for(var i = 0; i < this.inputFields.length; i++) {
		this.inputFields[i].setEditMode(false);
		var value = this.inputFields[i].displayValue;
		
		if(value) {
			if(i == 0) {
				var bold = document.createElement("strong");
				this.inputFields[i].appendTo(bold);
				this.displayValue.appendChild(bold);
			} else {
				this.displayValue.appendChild(document.createTextNode(value));
			}
			
			if(this.inputFields.length > 1 && i != (this.inputFields.length-1)) {
				this.displayValue.appendChild(document.createElement("br"));
			}
		}
	}
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.deleteField = function() {
	this.deleted = true;
	DOMUtil.emptyNode(this.rootNode);
	
	this.rootNode.parentNode.parentNode.style.display = "none";
	return false;
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.cancelEdit = function() {
	this.currentValue = this.originalValue;
	
	var emptyFields = 0;
		
	for(var i = 0; i < this.inputFields.length; i++) {
		this.inputFields[i].setEditMode(false);
		this.inputFields[i].cancelEdit();
		
		if(i > 0 && i < 6) {
			if(!this.inputFields[i].getValue() || (this.inputFields[i].getValue().replace(/^\s*|\s*$/g,"") == "")) {
				emptyFields++;
			}
		}
	}
	
	if(emptyFields == 5) {
		this.deleteField();
		return;
	}
	
	this.constructViewNode();
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.setDisabled = function(disabled) {
	for(var i = 0; i < this.inputFields.length; i++) {
		this.inputFields[i].setDisabled(disabled);
	}
	
	this.deleteButton.disabled = disabled;
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.saveEdit = function() {
	for(var i = 0; i < this.inputFields.length; i++) {
		this.inputFields[i].saveEdit();
		this.inputFields[i].setEditMode(false);
	}
	
	this.originalValue = this.currentValue;
	this.constructViewNode();
}

behaviour.gui.updateables.CompoundUpdateableField.prototype.getValue = function() {
	var output = new Object();
	
	if(this.deleted) {
		return output;
	}
	
	for(var i = 0; i < this.inputFields.length; i++) {
		//Log.info("ouput - " + this.inputFieldNames[i] + " = " + this.inputFields[i].getValue());
		output[this.inputFieldNames[i]] = this.inputFields[i].getValue();
	}
	
	return output;
}
