behaviour.gui.updateables.UpdateableKeywords = new Class.create();

// extends TitleBar
Object.extend(behaviour.gui.updateables.UpdateableKeywords.prototype, behaviour.gui.updateables.UpdateableField.prototype);

behaviour.gui.updateables.UpdateableKeywords.prototype.deletedKeywords;
behaviour.gui.updateables.UpdateableKeywords.prototype.keywordInput;

/**
 */
behaviour.gui.updateables.UpdateableKeywords.prototype.initialize = function(keywords) {
	this.currentValue = new Array();
	
	if(keywords && keywords.length) {
		for(var i = 0; i < keywords.length; i++) {
			this.currentValue[this.currentValue.length] = keywords[i]["keyword"];
		}
	}
	
	this.setRootNode(document.createElement("span"));
	this.addClass("UpdateableKeywords");
	
	this.originalValue = BehaviourUtil.cloneObject(this.currentValue);
	this.deletedKeywords = new Array(this.currentValue.length);
	this.displayValue = this.getWords();
	this.setEditMode(false);
}

behaviour.gui.updateables.UpdateableKeywords.prototype.createEditField = function() {
	this.empty();
	
	var keywords = document.createElement("p");
	keywords.appendChild(this.getEditableWords());
	this.appendChild(keywords);
	
	var addForm = DOMUtil.createNullForm();
	DOMUtil.addClass(addForm, "addKeywordForm");
	
	this.keywordInput = DOMUtil.createTextInputNode(null, {name: "addKeywordText", className: "addKeywordText", id: "addKeywordText"});
	addForm.appendChild(this.keywordInput);
	
	var keywordAddButton = DOMUtil.createSubmitInputNode(Language.get("generic",  "add"), {className: "addKeywordButton"});
	keywordAddButton.onclick = this.addKeyword.bindAsEventListener(this);
	addForm.appendChild(keywordAddButton);
	
	this.appendChild(addForm);
}

behaviour.gui.updateables.UpdateableKeywords.prototype.getWords = function() {
	var output = "";
	
	for(var i = 0; i< this.currentValue.length; i++) {
		if(!this.deletedKeywords[i]) {
			if(i > 0 && output != "") {
				output += ", ";
			}
			
			output += this.currentValue[i];
		}
	}

	return (output != "" ? output : Language.get("generic",  "nokeywords"));
}

behaviour.gui.updateables.UpdateableKeywords.prototype.getEditableWords = function() {
	var keywordsAdded = 0;
	var output = document.createDocumentFragment();
		
	for(var i = 0; i< this.currentValue.length; i++) {
		if(!this.deletedKeywords[i]) {
			if( keywordsAdded > 0) {
				output.appendChild(document.createTextNode(", "));
			}
			
			output.appendChild(document.createTextNode(this.currentValue[i]));
			output.appendChild(document.createTextNode(" "));
			
			var deleteLink = document.createElement("a");
			deleteLink.href = ".";
			deleteLink.appendChild(document.createTextNode("x"));
			deleteLink.title = Language.get("generic",  "deletethiskeyword");
			deleteLink.num = i;
			deleteLink.onclick = this.deleteKeyword.bindAsEventListener(this);
			DOMUtil.addClass(deleteLink, "keywordDeleteLink");
			
			output.appendChild(deleteLink);
			keywordsAdded++;
		}
	}
	
	return (keywordsAdded > 0 ? output : document.createTextNode(Language.get("generic",  "nokeywords")));
}

behaviour.gui.updateables.UpdateableKeywords.prototype.deleteKeyword = function(event) {
	var element = Event.element(event);
	
	this.deletedKeywords[element.num] = true;
	
	this.displayValue = this.getWords();
	this.render();
	
	return false;
}

behaviour.gui.updateables.UpdateableKeywords.prototype.addKeyword = function(event) {
	var keyword = BehaviourUtil.trim($(this.keywordInput).value);
	
	$(this.keywordInput).value = "";
	
	if(keyword == "") {
		return false;
	}
	
	var keywords = keyword.split(",");
	
	for(var i = 0; i < keywords.length; i++) {
		keywords[i] = BehaviourUtil.trim(keywords[i]);
		
		for(var n = 0; n < this.currentValue.length; n++) {
			if(this.currentValue[n] == keywords[i]) {
				keywords.splice(i, 1);
				i--;
				break;
			}
		}
	}
	
	for(var i = 0; i < keywords.length; i++) {
		if(BehaviourUtil.trim(keywords[i]) != "") {
			this.currentValue[this.currentValue.length] = keywords[i];
			this.deletedKeywords[this.deletedKeywords.length] = false;
		}
	}
	
	
	
	this.displayValue = this.getWords();
	this.render();
	
	return false;
}

behaviour.gui.updateables.UpdateableKeywords.prototype.getValue = function() {
	var output = new Array();
	
	for(var i = 0; i < this.currentValue.length; i++) {
		if(!this.deletedKeywords[i]) {
			output[output.length] = this.currentValue[i];
		}
	}

	return output;
}

behaviour.gui.updateables.UpdateableKeywords.prototype.saveEdit = function() {
	this.currentValue = this.getValue();
	this.deletedKeywords = new Array(this.currentValue.length);
}

behaviour.gui.updateables.UpdateableKeywords.prototype.cancelEdit = function() {
	this.currentValue = BehaviourUtil.cloneObject(this.originalValue);
	this.deletedKeywords = new Array(this.currentValue.length);
	this.displayValue = this.getWords();
	this.render();
}
