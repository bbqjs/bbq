behaviour.gui.table.LiveTable = new Class.create();

Object.extend(behaviour.gui.table.LiveTable.prototype, behaviour.gui.table.SortableTable.prototype);

behaviour.gui.table.LiveTable.prototype.dataLoaded;

/**
 * Supports the following options:
 * 
 * options: {
 * 		data: {
 * 			url: String,
 * 			method: String,
 * 			args: Object
 * 		}
 * }
 */
behaviour.gui.table.LiveTable.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.table.SortableTable.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	this.dataLoaded = false;
	
	var request = new behaviour.ajax.JSONRequest(this.controller, this.options.data.url, this._gotValues.bind(this));
	request.sendRequest(this.options.data.method ? this.options.data.method : 'get', this.options.data.args ? this.options.data.args : "");
}

/**
 * @private
 */
behaviour.gui.table.LiveTable.prototype._gotValues = function(serverResponse, json) {
	this.entities = json[this.options.data.entities];
	this.dataLoaded = true;
	
	if(this.entities.length > 0) {
		if(this.options.startSorted) {
			if(this.options.startSorted.direction && this.options.startSorted.direction == "DESC") {
				this.sortedByColumnIndex = this.options.startSorted.column ? this.options.startSorted.column : 0;
			}
			
			this.sortByIndex(this.options.startSorted.column ? this.options.startSorted.column : 0);
		} else {
			this.sortByIndex(0);
		}
	}
}

/**
 * 
 */
behaviour.gui.table.LiveTable.prototype.render = function() {
	if(this.dataLoaded) {
		var parentMethod = behaviour.gui.table.SortableTable.prototype.render;
		parentMethod.apply(this, arguments);
	}
}

/**
 * Do not call SortableTable.appendTo as it tries to sort our data which may not have loaded yet.
 * 
 * @param	{Node}	pageNode
 * @return	void
 */
behaviour.gui.table.LiveTable.prototype.appendTo = function(pageNode) {
	var parentMethod = behaviour.gui.GUIWidget.prototype.appendTo;
	parentMethod.apply(this, arguments);
}
