/**
 * @base behaviour.gui.button.GUIButton
 * @class
 * 
 * NativeButton
 * 
 * This class is similar to the GUIButton class but uses a standard browser submit input button as its
 * base. It supports the following options:
 * 
 * {
 * 		buttonText: string,	// the text contatined in the anchor tag and tool tip
 * 		onclick: function,		// onclick callback
 * 		onmouseover:	function,	// onmouseover callback
 * 		onmouseout:	 function,	// onmouseout callback
 * 		callbackObject: object,	// an object passed as an argument to the above callback functions
 * 		rememberDownState: boolean,	// whether or not to add the class buttonDown after clicking
 * 		buttonID: string,			// Unique HTML ID for button
 * 		startDisabled: boolean			// whether or not this button should start off disabled
 * }
 */
behaviour.gui.button.NativeButton = Class.create();

Object.extend(behaviour.gui.button.NativeButton.prototype, behaviour.gui.button.GUIButton.prototype);

behaviour.gui.button.NativeButton.prototype.submitButton;

/**
 * Supports the same options as behaviour.gui.button.GUIButton
 */
behaviour.gui.button.NativeButton.prototype.initialize = function(options) {
	var parentConstructor = behaviour.gui.GUIWidget.prototype.initialize;
	parentConstructor.apply(this, arguments);
	
	this.setRootNode(document.createElement("li"));
	this.addClass("NativeButton");
	
	this.submitButton = DOMUtil.createSubmitInputNode((options.buttonText ? options.buttonText : " "), {onclick: function(){this.blur(); return false}});
	this.appendChild(this.submitButton);
	
	if(options.startDisabled) {
		this.setDisabled(true);
	} else {
		this.setDisabled(false);
	}
	
	Event.observe(this.submitButton, "click", this.buttonClicked.bindAsEventListener(this));
}

behaviour.gui.button.NativeButton.prototype.setDisabled = function(disabled) {
	this.disabled = disabled;
	this.submitButton.disabled = this.disabled;
}

 * Overridden as it's not really necessary with this type of button
 */

}
