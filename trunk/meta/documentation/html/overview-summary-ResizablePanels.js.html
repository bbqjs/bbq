/**
 * @class
 */
behaviour.gui.panel.ResizeablePanels  = new Class.create();

Object.extend(behaviour.gui.panel.ResizeablePanels.prototype, behaviour.gui.panel.DualPanel.prototype);

behaviour.gui.panel.ResizeablePanels.prototype._resizer;

/**
 * @extends	behaviour.gui.panel.DualPanel
 */
behaviour.gui.panel.ResizeablePanels.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.panel.DualPanel.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	this.addClass("ResizeablePanels");
	this.addClass((this.options.direction == "horizontal" ? "Horizontal" : "Vertical") + "ResizeablePanels");
}

/**
 * Returns the resizer widget
 * 
 * @return	behaviour.gui.Resizer
 */
behaviour.gui.panel.ResizeablePanels.prototype.getResizer = function() {
	return this._resizer;
}

/**
 * Draws the panels
 * 
 * @return	void
 */
behaviour.gui.panel.ResizeablePanels.prototype.render = function() {
	this.empty();
	
	try {
		this._resizer = new behaviour.gui.Resizer({
			div1: this._getChildPanel("panel1").getRootNode ? this._getChildPanel("panel1").getRootNode() : this._getChildPanel("panel1"),
			div1Min: (this.options.panel1Min ? this.options.panel1Min : 200),
			div2: this._getChildPanel("panel2").getRootNode ? this._getChildPanel("panel2").getRootNode() : this._getChildPanel("panel2"),
			div2Min: (this.options.panel2Min ? this.options.panel2Min : 200),
			horizontal: this.options.direction == "horizontal"
		});
		this.appendChild(this._getChildPanel("panel1"));
		this.appendChild(this._resizer);
		this.appendChild(this._getChildPanel("panel2"));
	} catch(e) {
		Log.dumpObject(e);
	}	
}
