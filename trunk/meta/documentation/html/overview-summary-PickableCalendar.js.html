// Constructor
behaviour.gui.PickableCalendar.prototype.initialize = function(controller, dateController, title, startDate, startX, startY) {
	this.controller = controller;
	this.dateController = dateController;
	this.dateControllerCallback = this.dateController.datePicked.bind(this.dateController);
	
	if(startDate instanceof Date) {
		this.date = startDate;
	} else {
		this.date = new Date(startDate);
	}
	
	this.setRootNode(document.createElement("div"));
	DOMUtil.addClass(this.rootNode, "pickableCalendar");
	
	this.initialDate = new Date(this.date.getTime());
	this.title = title;
	this.render();
}

behaviour.gui.PickableCalendar.prototype.render = function() {
	// check for leap years
	if(this.date.getMonth() == 1 && (this.date.getFullYear() % 4 == 0 || (this.date.getFullYear() % 100 == 0 && this.date.getFullYear() % 400 == 0))) {
		this.monthDays[1] = 29;
	} else {
		this.monthDays[1] = 28;
	}
	
	DOMUtil.emptyNode(this.rootNode);
	
	var dateTable = document.createElement("table");
	var dateTableBody = document.createElement("tbody");
	dateTable.appendChild(dateTableBody);
	
	// title
	var titleRow = DOMUtil.createTableHeaderRow(1);
	titleRow.childNodes[0].appendChild(document.createTextNode(this.title));
	titleRow.childNodes[0].colSpan = 7;
	DOMUtil.addClass(titleRow.childNodes[0], "titlecell");
	dateTableBody.appendChild(titleRow);

	var topRow = DOMUtil.createTableHeaderRow(3);
	
	// back
	var backMonthLink = DOMUtil.createTextElement("a", "<", {href: ".", onclick: this.previousMonth.bindAsEventListener(this), title: this.controller.getLanguage("generic", "previousmonth")});
	DOMUtil.addClass(topRow.childNodes[0], "prevnext");
	topRow.childNodes[0].appendChild(backMonthLink);
	
	// month
	topRow.childNodes[1].appendChild(document.createTextNode(this.controller.getLanguage("generic", "date", "months-" + this.date.getMonth()) + " " + this.date.getFullYear()));
	topRow.childNodes[1].colSpan = 5;
	
	// forwards
	var forwardMonthLink = DOMUtil.createTextElement("a", ">", {href: ".", onclick: this.nextMonth.bindAsEventListener(this), title: this.controller.getLanguage("generic", "nextmonth")});
	DOMUtil.addClass(topRow.childNodes[2], "prevnext");
	topRow.childNodes[2].appendChild(forwardMonthLink);
	
	dateTableBody.appendChild(topRow);
	
	// day headers
	var dayRow = DOMUtil.createTableHeaderRow(7);
	
	for(var i = 0; i < 7; i++) {
		dayRow.childNodes[i].appendChild(document.createTextNode(this.controller.getLanguage("generic", "date", "days-" + i).substr(0, 1)));
	}
	
	dateTableBody.appendChild(dayRow);
	
	// pad out the first row
	var tempDate = new Date(this.date.getTime());
	tempDate.setDate(1);
	
	var padding = tempDate.getDay();
	var row = document.createElement("tr");
	
	var n = 0;
	
	for(; n < padding; n++) {
		var cell = document.createElement("td");
		cell.appendChild(document.createTextNode(" "));
		row.appendChild(cell);
	}
	
	// loop through the days of the month
	var j = 1;
	
	for(var k = 1; k <= this.monthDays[this.date.getMonth()]; k++) {
		if(n == 7) {
			dateTableBody.appendChild(row);
			row = document.createElement("tr");
			
			if(j % 2 == 0) {
				DOMUtil.addClass(row, "even");
			}
			
			n = 1;
			j++;
		} else {
			n++;
		}
		
		var cellLink = document.createElement("a");
		cellLink.appendChild(document.createTextNode(k));
		cellLink.href = ".";
		
		cellLink.onclick = this.datePicked.bindAsEventListener(this);
		
		var cell = document.createElement("td");
		cell.appendChild(cellLink);
		
		tempDate.setDate(k);
		
		DOMUtil.addClass(cell, "day");
		
		if(this.equals(tempDate, this.initialDate)) {
			DOMUtil.addClass(cell, "originalDate");
		} else if(this.equals(tempDate, new Date())) {
			DOMUtil.addClass(cell, "today");
		}
		
		row.appendChild(cell);
	}
	
	// pad out the last row if necessary
	if(n != 0) {
		for(; n < 7; n++) {
			var cell = document.createElement("td");
			cell.appendChild(document.createTextNode(" "));
			row.appendChild(cell);
		}
	
		dateTableBody.appendChild(row);
	}
	
	this.rootNode.appendChild(dateTable);
}

behaviour.gui.PickableCalendar.prototype.equals = function(date1, date2) {
	return (date1.getFullYear() == date2.getFullYear() && date1.getMonth() == date2.getMonth() && date1.getDate() == date2.getDate());
}

behaviour.gui.PickableCalendar.prototype.previousMonth = function(event) {
	var element = Event.element(event);
	element.blur();
	
	this.date.setMonth(this.date.getMonth() - 1);

	this.render();	

	return false;
}

behaviour.gui.PickableCalendar.prototype.nextMonth = function(event) {
	var element = Event.element(event);
	element.blur();
	
	this.date.setMonth(this.date.getMonth() + 1);
	this.render();
	
	return false;
}

behaviour.gui.PickableCalendar.prototype.datePicked = function(event) {
	var element = Event.element(event);
	
	this.date.setDate(element.innerHTML);
	Log.info("Date = " + this.date);
	this.dateControllerCallback(this.date);
	
	return false;
}
