/**
 * @extends	behaviour.ajax.AJAXRequest
 */
behaviour.ajax.JSONRequest = Class.create();

Object.extend(behaviour.ajax.JSONRequest.prototype, behaviour.ajax.AJAXRequest.prototype);

/**
 * @param	String		handlerName
 * @param	Object		args
 */
behaviour.ajax.JSONRequest.prototype._callHandler = function(handlerName, args) {
	var serverResponse = args[0];
	var json = new Object();
	
	try {
		json = serverResponse.responseText.evalJSON(true);
	} catch(e) {
		Log.dumpObject(e);
	}
	
	var parentMethod = behaviour.ajax.AJAXRequest.prototype._callHandler;
	parentMethod.call(this, handlerName, [serverResponse, json]);
}
