behaviour.lang.Iterator.prototype.next = function() {
	if(this.hasNext()) {
		var member = this.theObject[this.i];
		this.i++;
		return member;
	}
	
	return false;
}

behaviour.lang.Iterator.prototype.hasNext = function() {
	return (this.i < this.theObject.length);
}

	return this.i;
}

	return (this.i > 0 ? this.i - 1 : null);
}

}
