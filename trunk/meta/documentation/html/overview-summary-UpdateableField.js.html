behaviour.gui.updateables.UpdateableField = new Class.create();

// extends TitleBar
Object.extend(behaviour.gui.updateables.UpdateableField.prototype, behaviour.gui.GUIWidget.prototype);

/**
 * Supports the following options:
 * 
 * options: {
 * 		initialValue:	mixed
 * 		required:	boolean
 * 		shortenDisplayField:	boolean
 * }
 */
behaviour.gui.updateables.UpdateableField.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.GUIWidget.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	this.setRootNode(document.createElement("span"));
	
	this.shortenDisplayField = this.options.shortenDisplayField;
	this.required = this.options.required;
	this.editMode = false;
	
	if(typeof(this.options.initialValue) == "undefined" || this.options.initialValue == null) {
		this.options.initialValue = "";
	}
	
	this.displayValue = this.options.initialValue;
	this.currentValue = this.options.initialValue;
	this.originalValue = this.options.initialValue;
	
	this.addClass("UpdateableField");
}

// resets the internal field value to that prior editing
behaviour.gui.updateables.UpdateableField.prototype.cancelEdit = function() {
	this.currentValue = this.originalValue;
	this.displayValue = this.originalValue;
}

behaviour.gui.updateables.UpdateableField.prototype.createEditField = function() {
	Log.info("Non overridden createEditField() method called");
}

behaviour.gui.updateables.UpdateableField.prototype.createViewField = function() {
	if(this.displayValue && this.displayValue.length > 0) {
		var displayValue = this.displayValue;
		
		if(this.shortenDisplayField && this.displayValue.length > 20) {
			displayValue = this.displayValue.substr(0, 9) + "..." + this.displayValue.substr((this.displayValue.length - 8), 8); 
		}
		
		return document.createTextNode(displayValue);
	}
	
	return document.createTextNode(" ");
}

// returns the current field value
behaviour.gui.updateables.UpdateableField.prototype.getValue = function() {
	//Log.info("Current value = '" + this.currentValue + "' trimmed is '" + this.currentValue.replace(/^\s*|\s*$/g,"") + "' required is " + this.required);
	
	if(this.required && this.currentValue.replace(/^\s*|\s*$/g,"") == "") {
		this.addClass("error");
		throw "field needs value";
	}
	
	return this.currentValue;
}

// draws the field either as a form for editing or text for display
behaviour.gui.updateables.UpdateableField.prototype.render = function() {
	//var oldNode = this.rootNode;
	
	this.empty();
	
	if(this.editMode) { // create edit node
		this.appendChild(this.createEditField());
	} else { // create view node
		this.appendChild(this.createViewField());
	}

	this.checkForRequired();
}

// saves the current field value internally
behaviour.gui.updateables.UpdateableField.prototype.saveEdit = function() {
	this.displayValue = this.currentValue;
	this.originalValue = this.currentValue;
}

behaviour.gui.updateables.UpdateableField.prototype.setDisabled = function(disabled) {
	this.disabled = disabled;
	this.render();
}

behaviour.gui.updateables.UpdateableField.prototype.setEditMode = function(editMode) {
	this.editMode = editMode;
	this.render();
}

behaviour.gui.updateables.UpdateableField.prototype.updateLocalValue = function() {
	//Log.info("Updating local value - '" + this.currentValue + "' becoming '" + this.rootNode.value + "'");
	
	this.currentValue = this.rootNode.value;
	this.checkForRequired();
}

behaviour.gui.updateables.UpdateableField.prototype.checkForRequired = function() {
	if(this.required && this.rootNode && this.rootNode.className) {
		if(this.currentValue.replace(/^\s*|\s*$/g,"") == "") {
			this.addClass("error");
		} else {
			this.removeClass("error");
		}
	}
}
