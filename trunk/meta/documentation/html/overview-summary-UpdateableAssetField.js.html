behaviour.gui.updateables.UpdateableAssetField.prototype.receiveFiles = function(files) {
	this.currentValue = new Array();
	
	if(files.length == 1) {
		this.oldImage = this.image;
		this.imagePath = files[0].thumbpath;
		this.updateImage(files[0].thumbpath, files[0].thumbwidth, files[0].thumbheight, "");
		this.currentValue = files[0].filetransferid;
	} else {
		for(var i = 0; i < files.length; i++) {
			this.currentValue[i] = files[i].filetransferid;
		}
	}
	
	this.render();
}
