behaviour.lang.DataHolder.prototype.addElement = function(element) {
	var oldElement = false;//this.getElementWithKey(element[this.keyField]);
	
	if(oldElement) {
		//Log.info("Found item in _objectArray already with " + this.keyField + " = " + element[this.keyField] + ", updating it");
		
		for(var key in element) {
			oldElement[key] = element[key];
		}
	} else {
		//Log.info("Adding new item with " + this.keyField + " = " + element[this.keyField]);
		
		this.arrayIndexes["index-" + element[this.keyField]] = element;
		//Log.info(this.arrayIndexes["index-" + element[this.keyField]] + " for key " + element[this.keyField]);
		this._objectArray.push(element);
	}
}

behaviour.lang.DataHolder.prototype.removeElement = function(element) {
	for(var i = 0; i < this._objectArray.length; i++) {
		if(this._objectArray[i][this.keyField] == element[this.keyField]) {
			this._objectArray.splice(i, 1);
			this.arrayIndexes["index-" + element[this.keyField]] = null;
			i--;
		}
	}
}
