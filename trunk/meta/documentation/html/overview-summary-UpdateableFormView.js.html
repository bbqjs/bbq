behaviour.gui.updateables.UpdateableFormView = new Class.create();

// extends GUIWidget
Object.extend(behaviour.gui.updateables.UpdateableFormView.prototype, behaviour.gui.updateables.UpdateableFieldGroup.prototype);

behaviour.gui.updateables.UpdateableFormView.prototype.icon;
behaviour.gui.updateables.UpdateableFormView.prototype.hasIcon = false;

// Constructor
behaviour.gui.updateables.UpdateableFormView.prototype.initialize = function(saveURL) {
	var parentConstructor = behaviour.gui.updateables.UpdateableFieldGroup.prototype.initialize;
	parentConstructor.apply(this, arguments);

	this.setRootNode(document.createElement("form"));
	this.addClass(this.rootNode, "UpdateableFormView");
	
	this.saveURL = saveURL;
	this.invisibleFields = new Array();
	this.expandingFields = new Array();
	this.fieldTitles = new Array();
}

behaviour.gui.updateables.UpdateableFormView.prototype.addInvisibleField = function(fieldName, fieldData) {
	this.addField(fieldName, fieldData);
	this.invisibleFields[this.fields.length - 1] = true
}

behaviour.gui.updateables.UpdateableFormView.prototype.addVisibleField = function(fieldName, fieldData, fieldTitle) {
	this.addField(fieldName, fieldData);
	this.fieldTitles[this.fields.length - 1] = fieldTitle;
	this.invisibleFields[this.fields.length -1] = false;
}

behaviour.gui.updateables.UpdateableFormView.prototype.addExpandingField = function(fieldName, fieldData, fieldTitle) {
	this.addField(fieldName, fieldData);
	this.fieldTitles[this.fields.length - 1] = fieldTitle;
	this.invisibleFields[this.fields.length -1] = false;
}

behaviour.gui.updateables.UpdateableFormView.prototype.addIcon = function(fieldName, fieldData) {
	this.addField(fieldName, fieldData);
	this.icon = fieldData;
}

behaviour.gui.updateables.UpdateableFormView.prototype.sendValues = function(callback) {
	var args = this.getValuesAsString();
	
	var saveRequest = new behaviour.ajax.JSONRequest({url: this.saveURL, oncomplete: callback});
	saveRequest.sendRequest('post', args);
}

behaviour.gui.updateables.UpdateableFormView.prototype.render = function() {
	this.empty();
	
	if(this.icon) {
		this.icon.setStyle("cssFloat", "right");
		this.icon.appendTo(this.rootNode);
	}
	
	for(var i = 0; i < this.fields.length; i++) {
		if(this.invisibleFields[i] || this.fields[i] == this.icon) {
			continue;
		}
		
		this.appendChild(DOMUtil.createTextElement("label", this.fieldTitles[i]));
		
		if(this.fields[i]) {
			this.fields[i].appendTo(this.rootNode);
		}
	}
}
