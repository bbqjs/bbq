/**
 * Uses the SWFObject library to wrapper an embedded flash movie.
 */
behaviour.web.FlashEmbed = new Class.create();

Object.extend(behaviour.web.FlashEmbed.prototype, behaviour.gui.GUIWidget.prototype);

behaviour.web.FlashEmbed.prototype.flashObjectContainer;
behaviour.web.FlashEmbed.prototype.embedTag;

behaviour.web.FlashEmbed.flashCallbacks;

/**
 * Supports the following options:
 * 
 * options {
 * 		flashStub: string,				// This will be used as a global variable and MUST be unique to the page
 * 		appendTo: Node				// Where to append this embed
 * 		swf: {
 * 			path: string,					// Path to the flash movie
 * 			id: string,						// The ID of the flash movie
 * 			width: int,						// Width of the flash movie
 * 			height: int,						// Height of the flash movie
 * 			flashVars: {					// Variables to pass to the movie in key: value pairs
 * 				key: value,
 * 				...
 * 			}
 * 		}
 * 		
 * }
 */
behaviour.web.FlashEmbed.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.GUIWidget.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	if(typeof(behaviour.web.FlashEmbed.flashIDs) == "undefined") {
		behaviour.web.FlashEmbed.flashCallbacks = {};
	}
	
	behaviour.web.FlashEmbed.flashCallbacks[options.flashStub] = this._getFlash.bind(this);

	Log.info("Creating new FlashEmbed");
		
	this.flashObject = false;
	
	if(options.swf.path.search(/\?/) == -1) {
		options.swf.path += "?nocache=" + BehaviourUtil.generateGUID();
	} else {
		options.swf.path += "&nocache=" + BehaviourUtil.generateGUID();
	}
	
	var so = new SWFObject(options.swf.path, options.swf.id, options.swf.width, options.swf.height, options.swf.version ? options.swf.version : 8);
	so.addParam("allowScriptAccess", "always");
	so.addParam("swliveconnect", "true");
	
	if(options.swf.flashVars instanceof Object) {
		for(key in options.swf.flashVars) {
			so.addVariable(key, options.swf.flashVars[key]);
		}
	}
		
	so.write(options.appendTo);
}

behaviour.web.FlashEmbed.prototype._getFlash = function() {
	Log.warn("Getting flash object from DOM with ID " + this.options.swf.id);
	
	if(!this.flashObject) {
		if($(this.options.swf.id)) {
			this.flashObject = $(this.options.swf.id);
			Log.info("flash = " + $(this.options.swf.id).parentNode.innerHTML);
			
			if(this.options.onload && this.options.onload instanceof Function) {
				this.options.onload.call(this);
			}
		} else {
			Log.info("DOM node not loaded yet");
			setTimeout(this._getFlash.bind(this), 500);
		}
	}
}

behaviour.web.FlashEmbed.prototype.callFlashFunction = function(methodName, args) {
	Log.info("Calling flash function " + methodName + " with args " + Object.toJSON(args));
	
	if(!this.flashObject) {
		Log.error("method call on flashObject that does not exist");
		
		if($(this.options.swf.id)) {
			Log.warn("Flash object is in the page though");
			this.flashObject = $(this.options.swf.id);
		}
	}
	
	if(this.flashObject[methodName]) {
		Log.info("Found method " + methodName + " on Flash object - " + typeof(this.flashObject[methodName]));
		return this.flashObject[methodName](args);
	}
	
	Log.warn("Could not find method " + methodName + " on Flash object");
}
