/**
 * Uploads a single file and displays a thumbnail of it
 * 
 * Supports the following options:
 * 
 * options = {
 * 		imageSide								How big the biggest dimension of the thumbnail will be - eg. 128 for 128px
 * 		image 	{									Starting image (optional)
 * 			path,										Path to thumbnail
 * 			width,										thumbnail width
 * 			height,									thumbnail height
 * 			alt											alternative text for thumbnail
 * 			},
 * 		uploadImageTo	,					URL to POST uploaded files to // "/upload.tb?sid=" + this.controller.sid + "&iec=" + Math.random() + "&action=uploadAsset"
 * 		lang {										Object containing language translation for the GUI
 * 			chooseFile,
 * 			uploadingFile,
 * 			cancel,
 * 			}
 * 		}
 * 
 * It works by hiding an iFrame with a normal HTML form (what else) with an onchange callback set on the file upload field.  When the user clicks the "Upload file" button, the file field in the
 * hidden form is triggered, showing the browser file selection box.  When the user chooses the file, the onchange callback fires which submits the form.
 * 
 * The iFrame is then polled continously until it loads a new page, meaning either the upload has finished or an error has occured.  If the upload worked ok, it calls a function on the newly
 * loaded page to display a thumbnail of the uploaded file and record the upload ID.  Otherwise it displays an error message to the user.
 */

behaviour.gui.updateables.UpdateableSingleFileField = new Class.create();

// extends TitleBar
Object.extend(behaviour.gui.updateables.UpdateableSingleFileField.prototype, behaviour.gui.updateables.UpdateableField.prototype);

// Constructor
behaviour.gui.updateables.UpdateableSingleFileField.prototype.initialize = function(options)
	{
	Log.info("Creating new UpdateableSingleFileField");
	this.setRootNode(document.createElement("span"));
	DOMUtil.addClass(this.rootNode, "UpdateableSingleFileField");
	
	this.options = options;
	
	if(this.options.image.path)
		{
		this.oldImage = new Object();
		
		for(var key in this.options.image)
			{
			this.oldImage[key] = this.options.image[key];
			}
		
		this.updateImage(this.oldImage);
		
		this.addClass(this.rootNode, "withImage");
		}
	
	this.setEditMode(false);
	}

/**
 * Sets the passed image to be the current image displayed
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.updateImage = function(imageDetails)
	{
	this.imageNode = new Image(imageDetails.width, imageDetails.height);
	this.imageNode.src = imageDetails.path;
	this.imageNode.alt = imageDetails.alt;
	
	if(imageDetails.height < this.options.imageSide)
		{
		var padding = Math.ceil((this.options.imageSide - imageDetails.height) / 2);
		this.imageNode.style.marginTop = padding + "px";
		this.imageNode.style.marginBottom = padding + "px";
		}
	
	if(imageDetails.width < this.options.imageSide)
		{
		var padding = Math.ceil((this.options.imageSide - imageDetails.width) / 2);
		this.imageNode.style.marginLeft = padding + "px";
		this.imageNode.style.marginRight = padding + "px";
		}
	
	if(imageDetails.uploadKey)
		{
		this.currentValue = imageDetails.uploadKey;
		}
	}

/**
 * Creates the div to hold the image thumbnail
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype._createImageArea = function()
	{
	this.imageArea = document.createElement("div");
	this.imageArea.style.height = this.options.imageSide + "px";
	this.imageArea.style.width = this.options.imageSide + "px";
	
	if(this.imageNode)
		{
		this.imageNode.style.display = "none";
		this.imageArea.appendChild(this.imageNode);
		Effect.Appear(this.imageNode, {duration: 0.25});
		DOMUtil.addClass(this.imageArea, "fileuploaded");
		}
	else
		{
		DOMUtil.addClass(this.imageArea, "nofileuploaded");
		}
	
	this.rootNode.appendChild(this.imageArea);
	}

/**
 * Show the upload button
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.createEditField = function()
	{
	Log.info("Creating edit field");
	this.empty();
	this._createImageArea();
	Log.info("Created image area");
	
	var buttonForm = document.createElement("form");
	
	this.chooseFileButton = DOMUtil.createSubmitInputNode(this.options.lang.chooseFile, {className: "chooseImageButton"});
	this.chooseFileButton.disabled = true;
	buttonForm.appendChild(this.chooseFileButton);
	
	this.rootNode.appendChild(buttonForm);
	
	// create hidden iFrame
	this.iFrame = new Object();
	
	this.iFrame.node = document.createElement("iframe");
	this.iFrame.node.src = "/uit/js/uploadFile.html";
	this.iFrame.node.style.display = "none";
	
	//this.iFrame.node.style.height = "100px";
	
	this.iFrame.loadingTimeOut = setInterval(this.iFrameLoaded.bind(this), 500);
	this.rootNode.appendChild(this.iFrame.node);
	
	this.originalFileName = "";
	
	Log.info("Created edit field");
	}

/**
 * Polls the hidden iFrame to see if the page with the actual upload form has loaded
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.iFrameLoaded = function()
	{
	var iFrameDocument = DOMUtil.getIFrameDocument(this.iFrame.node);
	
	if(iFrameDocument && iFrameDocument.getElementById &&iFrameDocument.getElementById("fileuploadfield"))
		{
		clearInterval(this.iFrame.loadingTimeOut);
		this.iFrame.fileField = iFrameDocument.getElementById("fileuploadfield");
		this.iFrame.fileField.onchange = this.fileFieldChanged.bind(this);
		this.iFrame.form = iFrameDocument.getElementById("fileuploadform");
		this.iFrame.form.action = this.options.uploadImageTo;
		this.chooseFileButton.disabled = this.disabled;
		this.chooseFileButton.onclick = this.showFilePickerWindow.bindAsEventListener(this);
		}
	}

/**
 * Triggers the file browser chooser - silly IE dependant method

behaviour.gui.updateables.UpdateableSingleFileField.prototype.showFilePickerWindow = function(event)

	// FIXME - make this work in Firefox

	}

/**
 * Returns the file name of a file from the file field in the hidden page
 * 
 * This method will shorten the file name if it's too long
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.getFileName = function(assetLocation)
	{
	// \ for windows, / for everything else
	var fileLocation = assetLocation.split(assetLocation.search("/") ? "\\" : "/");
				
	var fileName = fileLocation[fileLocation.length - 1];
	return (fileName.length > 19 ? fileName.substr(0, 8) + "..." + fileName.substr(fileName.length - 8, 8) : fileName);
	}

/**
 * Submits the hidden form once the user has chosen a file
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.fileFieldChanged = function()
	{
	this.originalFileName = this.getFileName(this.iFrame.fileField.value);
	this.iFrame.form.submit();
	DOMUtil.emptyNode(this.imageArea);
	DOMUtil.removeClass(this.imageArea, "nofileuploaded");
	DOMUtil.addClass(this.imageArea, "uploadingFile");
	
	this.imageArea.appendChild(DOMUtil.createTextElement("p", this.options.lang.uploadingFile));
	
	this.chooseFileButton.value = this.options.lang.cancel;
	this.chooseFileButton.onclick = this.createEditField.bind(this);
	
	// start watching the iframe for the page refresh
	setTimeout(this.getUploadedData.bind(this), 2000);
	}

/**
 * Polls the hidden iFrame to see if the file has finished uploading.
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.getUploadedData = function()
	{
	Log.info("getting uploaded data");
	var finishedUploading = true;
	
	if(this.iFrame.node.contentWindow && this.iFrame.node.contentWindow.getData)
		{
		var iframeWindow = this.iFrame.node.contentWindow;
				
		if(iframeWindow.fileError)
			{
			Log.info("error uploading file");
			var output = iframeWindow.getData();
			this.fileUploadError();
			}
		else
			{
			Log.info("getting data");
			var output = iframeWindow.getData();
				
			for(var key in output)
				{
				Log.warn(key + " = " + output[key] );
				}
			
			this.fileUploaded({path: output.thumbpath, height: output.thumbheight, width: output.thumbwidth, alt: "", transferID: output.filetransferid, size: output.filesize});
			}
		}
	else
		{
		finishedUploading = false;
		}
	
	if(!finishedUploading && !this.windowClosed)
		{
		// not finished yet, try again in 2 seconds
		setTimeout(this.getUploadedData.bind(this), 1000);
		}
	}

/**
 * Shows an error message to the user as the file upload has failed for some reason
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.fileUploadError = function()
	{
	Log.error("Upload failed");
	}

/**
 * Shows a thumbnail of the uploaded file
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.fileUploaded = function(fileData)
	{
	DOMUtil.emptyNode(this.imageArea);
	
	this.updateImage(fileData);
	this.createEditField();
	
	//this.originalFileName
	
	Log.info("Upload suceeded");
	}

/**
 * Shows a non-editable version of the file
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.createViewField = function()
	{
	Log.info("Creating view field");
	this.empty();
	this._createImageArea();
	}

/**
 * Resets the field to the value passed in at the beginning
 */
behaviour.gui.updateables.UpdateableSingleFileField.prototype.cancelEdit = function()
	{
	this.imageNode = null;
	this.currentValue = null;

	if(this.oldImage)
		{
		this.updateImage(this.oldImage);
		}
	
	this.createViewField();
	}
