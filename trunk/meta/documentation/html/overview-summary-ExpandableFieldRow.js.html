// Constructor
behaviour.gui.updateables.ExpandableFieldGroup.prototype.initialize = function(title, newItemText, newItemMethod)
	{
	var parentConstructor = behaviour.gui.updateables.UpdateableFieldGroup.prototype.initialize;
	parentConstructor.call(this);
	
	this.title = title;
	
	this.rootNode = document.createElement("table");
	DOMUtil.addClass(this.rootNode, "expandableFieldGroup");
	
	this.tableHead = document.createElement("thead");
	this.tableBody = document.createElement("tbody");
	
	this.rootNode.appendChild(this.tableHead);
	this.rootNode.appendChild(this.tableBody);
	
	this.newItemMethod = newItemMethod;
	
	var headerRow = document.createElement("tr");
	var headerCell = document.createElement("th");
		
	//this.newItemButton = this.controller.domUtil.createSubmitInputNode(null, null, "", newItemText);
	//this.newItemButton = this.controller.domUtil.createSubmitInputNode(null, null, "", "+");
	this.newItemButton = DOMUtil.createTextElement("a", newItemText, {href: ".", onclick: this.addItem.bindAsEventListener(this), className: "addLink"});
	headerCell.appendChild(this.newItemButton);
	
	headerRow.appendChild(headerCell);
	this.tableHead.appendChild(headerRow);
	this.newItemButton.style.display = "none";
	}

behaviour.gui.updateables.ExpandableFieldGroup.prototype.render = function()
	{
	for(var i = 0; i < this.fields.length; i++)
		{
		this.fields[i].updateLocalValue();
		}
	
	DOMUtil.emptyNode(this.tableBody);
	
	for(var i = 0; i < this.fields.length; i++)
		{
		if(this.fields[i].deleted)
			{
			continue;
			}
		
		var row = document.createElement("tr");
		var cell = document.createElement("td");
		this.fields[i].appendTo(cell);
		row.appendChild(cell);
		this.tableBody.appendChild(row);
		
		this.fields[i].setEditMode(this.editMode);
		}
	}

behaviour.gui.updateables.ExpandableFieldGroup.prototype.setEditMode = function(editMode)
	{
	var parentMethod = behaviour.gui.updateables.UpdateableFieldGroup.prototype.setEditMode;
	parentMethod.call(this, editMode);
	
	this.editMode = editMode;
	this.newItemButton.style.display = (editMode ? "block" : "none");
	}

behaviour.gui.updateables.ExpandableFieldGroup.prototype.addItem = function()
	{
	var newItem = this.newItemMethod();
	this.addField(this.title + "-" + this.fields.length, newItem);
	
	var row = document.createElement("tr");
	var cell = document.createElement("td");
	newItem.appendTo(cell);
	row.appendChild(cell);
	this.tableBody.appendChild(row);
	
	newItem.setEditMode(this.editMode);
	
	return false;
	}

behaviour.gui.updateables.ExpandableFieldGroup.prototype.addPreExistingItem = function()
	{
	var newItem = this.newItemMethod.apply(this, arguments);
	this.addField(this.title + "-" + this.fields.length, newItem);
	}

behaviour.gui.updateables.ExpandableFieldGroup.prototype.getValue = function()
	{
	var output = "";
	var n = 0;
	
	for(var i = 0; i < this.fields.length; i++)
		{
		if(this.fields[i].deleted)
			{
			continue;
			}
		
		output += BehaviourUtil.serializeForPHP(this.fields[i].getValue());
		
		if(i != (this.fields.length - 1))
			{
			output += "|";
			}
		
		n++;
		}
	
	return n + "|" + output;
	}

behaviour.gui.updateables.ExpandableFieldGroup.prototype.setDisabled = function(disabled)
	{
	var parentMethod = behaviour.gui.updateables.UpdateableFieldGroup.prototype.setDisabled;
	parentMethod.apply(this, arguments);
	
	this.newItemButton.disabled = disabled;
	}
