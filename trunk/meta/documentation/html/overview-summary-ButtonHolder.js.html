/**
 * Holds behaviour.gui.button.GUIButtons
 */
behaviour.gui.button.ButtonHolder = Class.create();

Object.extend(behaviour.gui.button.ButtonHolder.prototype, behaviour.gui.GUIWidget.prototype);

behaviour.gui.button.ButtonHolder.prototype._selectedIndex;

behaviour.gui.button.ButtonHolder.prototype.initialize = function(options) {
	var parentConstructor = behaviour.gui.GUIWidget.prototype.initialize;
	parentConstructor.apply(this, arguments);
	
	this.setRootNode(document.createElement("ul"));
	this.addClass("ButtonHolder");
}

/**
 * @param	behaviour.gui.button.GUIButtons	button
 */
behaviour.gui.button.ButtonHolder.prototype.addButton = function(button) {
	button.setIndex(this.childWidgets.length);
	button.buttonHolder = this;
	this.appendChild(button);
}

/**
 * Removes the down state on all child buttons
 */
behaviour.gui.button.ButtonHolder.prototype.clearDown = function() {
	this.childWidgets.invoke("clearDown");
}

/**
 * Disables every child button
 */
behaviour.gui.button.ButtonHolder.prototype.setDisabled = function(disabled) {
	this.childWidgets.invoke("setDisabled", disabled);
}

/**
 * Returns a button
 * 
 * @param	integer	index
 */
behaviour.gui.button.ButtonHolder.prototype.getButton = function(index) {
	return this.childWidgets[index];
}

/**
 * Returns the index of the currently selected button
 * 
 * @return	integer
 */
behaviour.gui.button.ButtonHolder.prototype.getSelectedIndex = function() {
	return this._selectedIndex;
}

/**
 * Sets the currently selected button index
 */
behaviour.gui.button.ButtonHolder.prototype.setSelectedIndex = function(index) {
	this._selectedIndex = index;
}

behaviour.gui.button.ButtonHolder.prototype._buttonClicked = function(button) {
	this.setSelectedIndex(button.getIndex());
	
	if(FocusWatcher.getKeypressCallbackObject() != this) {
		FocusWatcher.setKeypressCallbackObject(this);
	}
}

behaviour.gui.button.ButtonHolder.prototype.acceptFocus = function() {
	this.addClass("hasFocus");
}

behaviour.gui.button.ButtonHolder.prototype.loseFocus = function() {
	this.removeClass("hasFocus");
}

behaviour.gui.button.ButtonHolder.prototype.processKeypress = function(event) {
	var nextEntity = false;
	var index = this.getSelectedIndex();
	
	if(index != -1) {
		// process keypress
		if(event.keyCode == Event.KEY_LEFT) { // previous button
			nextEntity = this.getButton(index - 1);
		} else if(event.keyCode == Event.KEY_RIGHT) {  // next button
			nextEntity = this.getButton(index + 1);
		}
	}
	
	if(nextEntity) {
		nextEntity.buttonClicked();
	}
	
	return true;
}
