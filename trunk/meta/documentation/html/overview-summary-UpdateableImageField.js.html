/**
 * Supports the following options:
 * 
 * options: {
 * 		path: String									// where the current image is stored (if any)
 * 		width:	int										// how wide the image is
 * 		height: int										// how high the image is
 * 		alt:	String										// for the alt tag
 * 		uploadTo										// Where to POST the image
 * 		language: Object							// passed to a behaviour.gui.FileUploader object
 * }
 */
behaviour.gui.updateables.UpdateableImageField.prototype.initialize = function(options) {
	var parentMethod = behaviour.gui.GUIWidget.prototype.initialize;
	parentMethod.apply(this, arguments);
	
	this.setRootNode(document.createElement("span"));
	this.addClass("UpdateableImageField");
	
	if(this.options.path) {
		this.oldImagePath = this.options.path;
		this.oldImageWidth = this.options.width;
		this.oldImageHeight = this.options.height;
		this.oldImageAlt = this.options.alt;
		this.updateImage(this.options.path, this.options.width, this.options.height, this.options.alt);
		
		this.addClass("withImage");
	}
	
	this.width = this.options.width;
	this.height = this.options.height;
	
	this.setEditMode(false);
}

behaviour.gui.updateables.UpdateableImageField.prototype.updateImage = function(imagePath, width, height, alt) {
	this.image = false;
	
	if(typeof(imagePath) != "undefined") {
		this.image = new Image(width, height);
		this.image.src = imagePath;
		this.image.alt = alt;
		this.width = width;
		this.height = height;
		
		if(height < 140) {
			var padding = Math.ceil((140 - height) / 2);
			this.image.style.marginTop = padding + "px";
			this.image.style.marginBottom = padding + "px";
		}
	}
}

behaviour.gui.updateables.UpdateableImageField.prototype.createEditField = function() {
	this.empty();
	var form = DOMUtil.createNullForm();
	
	if(this.image) {
		this.appendChild(this.image);
		this.addClass("withImage");
		this.setStyle("paddingTop", "0px");
	}
	else {
		this.removeClass("withImage");
		this.setStyle("paddingTop", "120px");
	}
	
	var chooseImageButton = DOMUtil.createSubmitInputNode(Language.get("generic",  "chooseimage"), {className: "chooseImageButton", onclick: this.showFilePickerWindow.bindAsEventListener(this), disabled: this.disabled});
	form.appendChild(chooseImageButton);
	
	this.appendChild(form);
}

behaviour.gui.updateables.UpdateableImageField.prototype.createViewField = function() {
	this.empty();
	
	if(this.image) {
		this.appendChild(this.image);
		this.addClass("withImage");
		this.setStyle("paddingTop", "0px");
	} else {
		this.removeClass("withImage");
		this.setStyle("paddingTop", "120px");
	}
}

behaviour.gui.updateables.UpdateableImageField.prototype.showFilePickerWindow = function(event) {
	var imagePicker = new behaviour.gui.FileUploader({owner: this, multiple: false, sendFilesTo: this.options.uploadTo, language: this.options.language, acceptedFileExtensions: ["jpg", "gif", "png"]});
	currentPage.addModalLayer(imagePicker);
	
	return false;
}

behaviour.gui.updateables.UpdateableImageField.prototype.receiveFiles = function(files) {
	this.oldImage = this.image;
	this.imagePath = files[0].thumbpath;
	this.updateImage(files[0].thumbpath, files[0].thumbwidth, files[0].thumbheight, "");
	
	this.currentValue = files[0].filetransferid;
	this.render();
}

behaviour.gui.updateables.UpdateableImageField.prototype.cancelEdit = function() {
	this.image = null;
	this.currentValue = null;

	this.updateImage(this.oldImagePath, this.oldImageWidth, this.oldImageHeight, this.oldImageAlt);
	this.createViewField();
}

behaviour.gui.updateables.UpdateableImageField.prototype.saveEdit = function() {
	this.oldImagePath = this.imagePath;
	this.oldImageWidth = this.width;
	this.oldImageHeight = this.height;
}

behaviour.gui.updateables.UpdateableImageField.prototype.getValue = function() {
	return this.currentValue;
}
