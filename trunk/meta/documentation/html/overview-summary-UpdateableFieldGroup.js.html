	this.fields[this.fields.length] = field;

}

behaviour.gui.updateables.UpdateableFieldGroup.prototype.getValues = function() {
	var output = new Object();
	
	for(var i = 0; i < this.fields.length; i++) {
		if(this.fields[i]) {
			output[this.fieldNames[i]] = this.fields[i].getValue();
		}
	}
	
	return output;
}

behaviour.gui.updateables.UpdateableFieldGroup.prototype.getValuesAsString = function() {
	var output = "";
	var values = this.getValues();
	
	for(var key in values) {
		output += this.parseValues(key, values[key]);
	}
	
	return output;
}

behaviour.gui.updateables.UpdateableFieldGroup.prototype.parseValues = function(key, value) {
	if(value instanceof Array) {
		for(var i = 0; i < value.length; i++) {
			output += this.parseValues(key + "[]", value[i]) + "&";
		}
	} else if(key) {
		output = key + "=" + encodeURIComponent(value) + "&";
	}
	
	return output;
}

behaviour.gui.updateables.UpdateableFieldGroup.prototype.getValue = function(key) {
	for(var i = 0; i < this.fieldNames.length; i++) {
		if(this.fieldNames[i] == key) {
			return this.fields[i].getValue();
		}
	}
}
