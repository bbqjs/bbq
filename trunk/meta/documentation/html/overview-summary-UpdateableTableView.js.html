behaviour.gui.updateables.UpdateableTableView.prototype.initialize = function(options) {
	var parentConstructor = behaviour.gui.updateables.UpdateableFieldGroup.prototype.initialize;
	parentConstructor.apply(this, arguments);

	this.setRootNode(document.createElement("table"));
		
	this.addClass("UpdateableTableView");
	this.tableBody = document.createElement("tbody");
	this.appendChild(this.tableBody);
	
	this.saveURL = this.options.saveURL;
	this.invisibleFields = new Array();
	this.expandingFields = new Array();
	this.fieldTitles = new Array();
	this.withStyledFirstRow = this.options.withStyledFirstRow;
}

behaviour.gui.updateables.UpdateableTableView.prototype.addInvisibleRow = function(fieldName, fieldData) {
	this.addField(fieldName, fieldData);
	
	//Log.info("Added invisible field, fields.length = " + this.fields.length);
	
	this.invisibleFields[this.fields.length - 1] = true
}

behaviour.gui.updateables.UpdateableTableView.prototype.addRow = function(fieldName, fieldData, fieldTitle) {
	this.addField(fieldName, fieldData);
	this.fieldTitles[this.fields.length - 1] = fieldTitle;
	this.invisibleFields[this.fields.length -1] = false;
}

behaviour.gui.updateables.UpdateableTableView.prototype.addExpandingRow = function(fieldName, fieldData, fieldTitle) {
	this.addField(fieldName, fieldData);
	this.fieldTitles[this.fields.length - 1] = fieldTitle;
	this.invisibleFields[this.fields.length -1] = false;
}

behaviour.gui.updateables.UpdateableTableView.prototype.addIconRow = function(fieldName, fieldData) {
	Log.info("adding icon row");
	this.addField(fieldName, fieldData);
	this.icon = fieldData;
	Log.info("added icon row");
}

behaviour.gui.updateables.UpdateableTableView.prototype.sendValues = function(callback) {
	var args = this.getValuesAsString();
	
	var saveRequest = new behaviour.ajax.JSONRequest({url: this.saveURL, oncomplete: callback});
	saveRequest.sendRequest('post', args);
}

behaviour.gui.updateables.UpdateableTableView.prototype.render = function() {
	DOMUtil.emptyNode(this.tableBody);
	
	var rowCount = 0;
	var iconRowCount = -1;
	
	for(var i = 0; i < this.fields.length; i++) {
		//Log.info("name = " + this.fieldNames[i] + " title = " + this.fieldTitles[i] + " invisible = " + this.invisibleFields[i]);
		
		if(this.invisibleFields[i] || this.fields[i] == this.icon) {
			continue;
		}
		
		var row = document.createElement("tr");
		
		this.stripe(rowCount, row);
		
		var header = document.createElement("th");
		header.appendChild(document.createTextNode(this.fieldTitles[i]));
		row.appendChild(header);
		
		var cell = document.createElement("td");
		
		if(this.fields[i]) {
			this.fields[i].appendTo(cell);
		}
		
		row.appendChild(cell);
		
		this.tableBody.appendChild(row);
		
		// test for icon
		if(this.icon) {
			if(iconRowCount == -1) {
				var iconCell = document.createElement("td");
				DOMUtil.addClass(iconCell, "iconCell");
				this.icon.appendTo(iconCell);			
				iconCell.rowSpan = "6";
				
				iconCell.width = (this.icon.width ? this.icon.width : 120);
				iconCell.style.width = (this.icon.width ? this.icon.width : 120) + "px";
				iconCell.vAlign = "top";
				row.appendChild(iconCell);
				
				iconRowCount = 0;
			}
			
			if(iconRowCount >= 6) {
				cell.colSpan = "2";
				/*var extraCell = document.createElement("td");
				extraCell.appendChild(document.createTextNode(""));
				row.appendChild(extraCell);*/
			}
			
			iconRowCount++;
		}
		
		rowCount++;
	}
	
	if(this.icon) {
		while(iconRowCount < 6) { // pad table with extra rows so is valid
			var row = document.createElement("tr");
			this.stripe(rowCount, row);
			row.appendChild(document.createElement("td"));
			row.appendChild(document.createElement("td"));
			this.tableBody.appendChild(row);
			rowCount++;
			iconRowCount++;
		}
	}
}

behaviour.gui.updateables.UpdateableTableView.prototype.stripe = function(rowCount, row) {
	if(this.withStyledFirstRow) {
		if(rowCount == 0) {
			DOMUtil.addClass(row, "first");
		} else if(rowCount % 2 == 0) {
			DOMUtil.addClass(row, "even");
		}
	} else if(rowCount % 2 == 1) {
		DOMUtil.addClass(row, "even");
	}
}
