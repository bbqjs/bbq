behaviour.gui.FloatingWindow = new Class.create();

Object.extend(behaviour.gui.FloatingWindow.prototype, behaviour.gui.GUIWidget.prototype);

behaviour.gui.FloatingWindow.topZIndex = 100;

behaviour.gui.FloatingWindow.prototype.contentDiv;

// Constructor
behaviour.gui.FloatingWindow.prototype.initialize = function(controller, windowController, startX, startY)
	{
	this.controller = controller;
	this.windowController = windowController;
	
	this.rootNode = document.createElement("div");
	DOMUtil.addClass(this.rootNode, "floatingWindow");
	this.rootNode.style.position = "absolute";
	this.rootNode.style.left = startX + "px";
	this.rootNode.style.top = startY + "px";
	
	behaviour.gui.FloatingWindow.topZIndex++;
	this.rootNode.style.zIndex = behaviour.gui.FloatingWindow.topZIndex;
	
	this.rootNode.style.display = "none";
	
	this.headerDiv = document.createElement("div");
	DOMUtil.addClass(this.headerDiv, "floatingWindowHeader");
	
	this.contentDiv = document.createElement("div");
	DOMUtil.addClass(this.contentDiv, "floatingWindowContent");
	
	this.rootNode.appendChild(this.headerDiv);
	this.rootNode.appendChild(this.contentDiv);
	
	var table = document.createElement("table");
	DOMUtil.addClass(table, "chrome");
	
	var row = DOMUtil.createTableRow(2);
		
	var dragLink = DOMUtil.createTextElement("a", this.controller.getLanguage("generic", "clickandholdtodrag"), {href: ".", onclick: this.blurIt.bindAsEventListener(this), onmousedown: this.startDrag.bindAsEventListener(this)});
	row.childNodes[0].appendChild(dragLink);
	DOMUtil.addClass(row.childNodes[0], "dragLinkCell");
	
	var closeLink = DOMUtil.createTextElement("a", this.controller.getLanguage("generic", "close"), {href: ".", onclick: this.closeWindow.bindAsEventListener(this)});
	row.childNodes[1].appendChild(closeLink);
	DOMUtil.addClass(row.childNodes[1], "closeLinkCell");
	
	table.appendChild(row);
	
	this.headerDiv.appendChild(table);
	}

behaviour.gui.FloatingWindow.prototype.setContent = function(content)
	{
	if(content)
		{
		if(content.appendTo)
			{
			content.appendTo(this.contentDiv);
			}
		else
			{
			this.contentDiv.appendChild(content);
			}
		}
	}

behaviour.gui.FloatingWindow.prototype.render = function()
	{
	if(this.content && this.content.render)
		{
		this.content.render();
		}
	}

behaviour.gui.FloatingWindow.prototype.appendTo = function(node)
	{
	var parentMethod = behaviour.gui.GUIWidget.prototype.appendTo;
	parentMethod.call(this, node);
	
	Effect.Appear(this.rootNode, { duration: 0.1 });
	}

behaviour.gui.FloatingWindow.prototype.closeWindow = function(event)
	{
	BehaviourUtil.clearFocus(event);
	this.windowController.windowClosed();
	
	Effect.Fade(this.rootNode, { duration: 0.25 });
	setTimeout(this.removeWindow.bind(this), 260)
		
	return false;
	}

behaviour.gui.FloatingWindow.prototype.removeWindow = function()
	{
	var parentNode = this.rootNode.parentNode;
	
	if(parentNode)
		{
		parentNode.removeChild(this.rootNode);
		}
	}

behaviour.gui.FloatingWindow.prototype.blurIt = function(event)
	{
	var element = Event.element(event);
	element.blur();
	return false;
	}

behaviour.gui.FloatingWindow.prototype.startDrag = function(event)
	{
	var element = Event.element(event);
	element.blur();
	
	behaviour.gui.FloatingWindow.topZIndex++;
	this.rootNode.style.zIndex = behaviour.gui.FloatingWindow.topZIndex;
	
	document.onmousemove = this.doDrag.bindAsEventListener(this);
	document.onmouseup = this.stopDrag.bindAsEventListener(this);
	}

behaviour.gui.FloatingWindow.prototype.stopDrag = function()
	{
	document.onmousemove = null;
	document.onmouseup = null;
	}

behaviour.gui.FloatingWindow.prototype.doDrag = function(event)
	{
	this.rootNode.style.left = Event.pointerX(event) - 10 + "px";
	this.rootNode.style.top = Event.pointerY(event) - 10 + "px";
	}
