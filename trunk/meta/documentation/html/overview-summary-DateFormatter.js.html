DateFormatter = {
	_date: null,
	_dateFormatter: null,
	_dateSeperator: null,
	_timeSeperator: null,
	
	setDateFormatter: function(dateFormatter) {
		this._dateFormatter = dateFormatter;
	},
	
	getFormattedDate: function(date) {
		var backupDate = new Date(this._date);
		this._date = date;
		var output = this.toString();
		this._date = backupDate;
		
		return output;
	},
	
	/**
	 * @return	{String}
	 */
	toString: function() {
		return this._dateFormatter.toString();
	},
	
	/**
	 * Returns something like:
	 * 
	 * 23/8/2007 13:40
	 * 
	 * @return	String
	 */
	getLongDate: function() {
		return this._dateFormatter.getLongDate();
	},
	
	/**
	 * Returns somthing like
	 * 
	 * 23/8/2007
	 * 
	 * @return	String
	 */
	getDate: function() {
	 	return this._dateFormatter.getDate();
	},
	
	/**
	 * Returns something like
	 * 
	 * 13:40
	 * 
	 * @return	String
	 */
	getTime: function() {
	 	return this._dateFormatter.getTime();
	},
	
	getOrder: function() {
	 	return this._dateFormatter.getOrder();
	},
	
	getDateSeperator: function() {
	 	return this._dateFormatter.getDateSeperator();
	},
	
	getTimeSeperator: function() {
	 	return this._dateFormatter.getTimeSeperator();
	}
}
