behaviour.gui.SortableGUIWidget.prototype.ascSortFunction = function (a, b) {
	var aValue = (a[this.currentSortKey] instanceof Function ? a[this.currentSortKey].call(this, a) : a[this.currentSortKey]);
	var bValue = (b[this.currentSortKey] instanceof Function ? b[this.currentSortKey].call(this, b) : b[this.currentSortKey]);
	
	if(typeof(aValue) != "undefined") {
		if(aValue === true || aValue === false) {
			return this.sortBool(aValue, bValue);
		} else if(aValue instanceof Array) {
			return this.sortArray(aValue, bValue)
		} else if(aValue.toLowerCase) {
			return this.sortString(aValue, bValue);
		} else {
			return aValue - bValue;
		}
	} else {
		var key = this.currentSortKey.split(".");
		
		if(aValue === true || a[this.currentSortKey] === false) {
			return this.sortBool(a[this.currentSortKey], b[this.currentSortKey]);
		} else if(a[key[0]][key[1]] instanceof Array) {
			return this.sortArray(a[key[0]][key[1]], b[key[0]][key[1]])
		} else if(a[key[0]][key[1]].toLowerCase) {
			return this.sortString(a[key[0]][key[1]], b[key[0]][key[1]]);
		} else {
			return a[key[0]][key[1]] - b[key[0]][key[1]];
		}
	}
}

behaviour.gui.SortableGUIWidget.prototype.sortString = function(a, b, asc) {
	if(b.toLowerCase() > a.toLowerCase()) {
		return (asc ? -1 : 1);
	} else if(b.toLowerCase() < a.toLowerCase()) {
		return (asc ? 1 : -1);
	} else {
 		return 0;
 	}
}

behaviour.gui.SortableGUIWidget.prototype.sortArray = function(a, b) {
	if(b.length > a.length) {
		return -1;
	} else if(b.length < a.length) {
		return 1;
	} else {
 		return 0;
 	}
}
